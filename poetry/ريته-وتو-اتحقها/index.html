الشطر الأولالشطر الثاني
رِيــــتــه وتـــو اتــحــقها قــانــصّهوتقول كــنــت انــريــدها إلا لــوّ
مــا جــابها اخريفيش صاعب نصّهيــنــوّض الــحرشا اتــقول هــيا تــوّ
مــواريث جــده وهو عليها موصّىولا خــير فــيه ان عاف مــا سمّو
وتــابــع ادروبـــه كــلها مــاصصّههــضاك كــان هــو وهــضا هو هوّ
مـــا قــالها مــن قــبل مــا شــخصّهكــي عــيت ( حــيها ) لمّدو وحشوّ
وراك تــعرفه قصدي وهذيك القصّهومــن هو اللي جفّل ومن ( شاكوّ )
وزاد تــعرفه مــن هــو اللي نغصّهزمــرة اكــلاب القاف عيت ( الشوّ)
يــجو كــلمتين ويــغلبو ف الــرصّهمــاريــت فــيــهن م الــمــواير قــوّ
ولــو الاصــمعي هضاك مستخلصّهخــبــرة تــجــارب مـــن زمــانه أوّ
عــــقــل رــــبي خــالــقه يــقــلصّهفـــي شــطر شــعري او بــما أبــدَوّ
انـــه يــمتلك اعــين فــقط تــمتصّهمـــن ســمح مــا قــالو ومــا خــطَوّ
انــا أؤكــد لــك بــانك لــو اتمحصّهســاتــقول لـــي بــالــك بــها جــلَّوّ
وانـــه شــبيه الــماس نــادر فصّهخــافي عــلي دوج الــعرب والضوّ
ومـــا تــرجمته بــلتاجي امــخلصّهمـــن كـــل شــائبة تــحتوي الــسوّ
ولــو كــان مــا تعرف انه مخمصّهوانـــه قــايــل ووصــلتم ومــا ردَوّ
الــرمحي ومــا يرجى اللي يرخصّهشــــعــارهــا ونــقــادهــا والــــجــوّ
وحرصا على التوضيح ماهي حصّهفــي لــغة ضــاد الــلي لــها احــتمَوّ
نــردو لــموضوعي الــلي حــمصّهاكــبــودهم بــما راعــو ومــا ابــتلَوّ
بــارماق دامــي ف الوغى مختصّهعــارفــينها جــتــهم بــهــا انـــرزوّ
واضــع بــها دســتورها امــخصصّهمـــا يــرومــها الا مَــن لــها ادّعــوّ
